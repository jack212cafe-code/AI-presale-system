```javascript
export async function getConversationsByProject(projectId) {
  const client = getSupabaseAdmin();

  if (!client) {
    return [];
  }

  const { data, error } = await client
    .from("conversations")
    .select("id, project_id, stage, created_at, updated_at")
    .eq("project_id", projectId)
    .order("created_at", { ascending: false });

  if (error) {
    throw new Error(`Failed to list conversations: ${error.message}`);
  }

  return data ?? [];
}
```

This follows the exact same pattern as `getConversationById` and `getMessagesByConversation` — uses `getSupabaseAdmin()`, returns empty array when no client, throws on error.
