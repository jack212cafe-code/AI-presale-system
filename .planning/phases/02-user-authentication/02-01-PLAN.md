From lib/admin-auth.js (SESSION PATTERN TO REPLICATE):
```javascript
const SESSION_COOKIE = "ai_presale_admin_session";
const SESSION_TTL_MS = 1000 * 60 * 60 * 12;
const sessions = new Map();

export function createAdminSession() {
  cleanupExpiredSessions();
  const token = randomUUID();
  sessions.set(token, { createdAt: Date.now(), expiresAt: Date.now() + SESSION_TTL_MS });
  return token;
}

export function isAuthenticatedAdminRequest(request) {
  cleanupExpiredSessions();
  const token = getAdminSessionToken(request);
  if (!token) return false;
  const session = sessions.get(token);
  if (!session) return false;
  session.expiresAt = Date.now() + SESSION_TTL_MS; // sliding expiry
  return true;
}

export function buildAdminSessionCookie(token) {
  return `${SESSION_COOKIE}=${encodeURIComponent(token)}; HttpOnly; SameSite=Lax; Path=/; Max-Age=${Math.floor(SESSION_TTL_MS / 1000)}`;
}
```
