    Add a new async function `handleDiscoveryQuestions({ conversation, conversationId, message })` following the exact same pattern as `handleAwaitingSelection`:

    ```javascript
    async function handleDiscoveryQuestions({ conversation, conversationId, message }) {
      return withTimeout(async () => {
        await addMessage(conversationId, "user", message);

        const project = await getProjectById(conversation.project_id);
        const intake = project.intake_json;

        const requirements = await runDiscoveryAgent(intake, {
          projectId: conversation.project_id,
          mode: "parse_answers",
          discoveryReply: message
        });
        await persistRequirementsJson(conversation.project_id, requirements);

        // D-17: build assumption announcement if defaults were applied
        let assumptionPrefix = "";
        if (requirements.assumptions_applied?.length > 0) {
          const items = requirements.assumptions_applied.join(", ");
          assumptionPrefix = `ผมจะใช้ค่ามาตรฐานสำหรับข้อมูลที่ยังไม่ได้ระบุ: ${items} — และจะระบุ assumption เหล่านี้ในเอกสาร proposal ด้วยครับ\n\n`;
          await addMessage(conversationId, "assistant", assumptionPrefix.trim());
        }

        // D-18: proceed to solution generation automatically
        const solution = await runSolutionAgent(requirements, { projectId: conversation.project_id });
        await persistSolutionJson(conversation.project_id, solution);

        await updateConversationStage(conversationId, "awaiting_selection");

        const solutionText = formatSolutionOptions(solution);
        const responseText = assumptionPrefix + solutionText;
        await addMessage(conversationId, "assistant", solutionText);

        return {
          conversation_id: conversationId,
          project_id: conversation.project_id,
          stage: "awaiting_selection",
          text: responseText,
          created: false
        };
      });
    }
    ```
